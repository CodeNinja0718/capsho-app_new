import { getTemplateByType } from '@/services/templates'
import { ref, watch, readonly, onMounted, computed } from 'vue'
import { useUser } from '@/composables/useUser'
import { backticksTemplate } from '@/utils'
import { useEngageCaptions } from './useEngageCaptions'
import { useUsersData } from '@/composables/useUsersData'
import store from '@/store'
/**
 *
 * @returns {{template: *, error: *, loading: *}}
 */

export function useTemplate() {
	const error = ref(null)
	const loading = ref(true)
	const template = computed(() => store.getters['app/getTemplates'])
	const { userData, user } = useUser()
	const { engageCaptions } = useEngageCaptions()
	const { sets } = useUsersData()

	const getTemplate = ({ templateName, type }) => {
		loading.value = true
		getTemplateByType({ template: templateName, type })
			.then(({outputs}) => {
				store.commit('app/setTemplates', outputs)
				setTemplatesText()
			})
			.catch((e) => error.value = e)
			.finally(() => loading.value = !loading.value)
	}

	const setTemplatesText = () => {
		for (let i = 0; i < template.value.length; i++) {
			const possibles = []
			engageCaptions.value.forEach((caption) => {
				if (template.value[i].type === caption.type || template.value[i].tool === caption.tool) {
					if (sets.value.includes(caption.set)) {
						possibles.push(caption.text)
					} else if (caption.tool === 'engage') {
						const temp = backticksTemplate(caption.text)
						possibles.push(temp)
					}
				}
			})
			if (possibles.length > 0) {
				const rand = Math.floor(Math.random() * possibles.length)
				template.value[i].text = possibles[rand]
			}
		}
	}

	watch(userData, (newUserData) => {
		if (newUserData) {
			getTemplate({
				templateName: user.value.template,
				type: user.value.type
			})
		}
	})

	onMounted(() => {
		getTemplate({
			templateName: user.value.template,
			type: user.value.type
		})
	})

	return {
		template,
		error: readonly(error),
		loading: readonly(loading)
	}
}
