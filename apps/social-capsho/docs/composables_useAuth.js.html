import { computed, ref, readonly } from 'vue'
import { getAuth, onAuthStateChanged, sendPasswordResetEmail, signOut } from 'firebase/auth'
import store from '@/store'
import router from '@/router'
import { getUserInfo } from '@/services/authService'

/**
 *  Firebase auth composable with an observer
 *
 * @returns {{resetPassword: function, isAuthenticated: ComputedRef<boolean>, authUser: ref, logOut: logOut}}
 */
export function useAuth() {
  const auth = getAuth()

  const authUser = ref(auth.currentUser)
  const isAuthenticated = computed(() => !!authUser.value)

  const logOut = () => {
    signOut(auth)
      .then(async () => {
        sessionStorage.clear()
        store.commit('app/logOut')
        await router.push({ name: 'Home' })
      }).catch((err) => {
      console.error(err)
      alert(err.message)
    })
  }

  const resetPassword = () => {
    sendPasswordResetEmail(auth, authUser.value.email)
      .then(async () => {
        if (authUser.value) {
          logOut()
        } else {
          await router.push({ name: 'SignIn' })
        }
      }).catch((err) => {
      console.error(err)
    })
  }

  onAuthStateChanged(auth, (user) => {
    if (user) {
      authUser.value = user
      getUserInfo()
        .then(() => {
          store.commit('app/logIn', true)
          store.commit('app/setUserData', {
            displayName: user.displayName,
            email: user.email,
            photoURL: user.photoURL,
            emailVerified: user.emailVerified,
            id: user.uid
          })
      })
    } else {
      store.commit('app/setUserData', null)
    }
  })

  return {
    isAuthenticated,
    authUser: readonly(authUser),
    logOut,
    resetPassword
  }
}
