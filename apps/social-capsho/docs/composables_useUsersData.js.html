import { getUserData } from '@/services/usersData'
import { ref, readonly, watch } from 'vue'
import { useUser } from '@/composables/useUser'


/**
 *
 * @returns {{emails: *, sets: *, emailSubjectLines: *, loading: *, error, captions: *}}
 */

export function useUsersData() {
	const sets = ref([])
	const captions = ref([])
	const emailSubjectLines = ref([])
	const emails = ref([])
	const error = ref(null)
	const loading = ref(true)

	const { userData } = useUser()

	const getCollectionData = () => {
		loading.value = true
		getUserData(userData.value.id)
			.then((doc) => {
				sets.value = doc.sets
				captions.value = doc.captions
				emailSubjectLines.value = doc.email_subject_lines
				emails.value = doc.emails
			})
			.catch((e) => error.value = e)
			.finally(() => loading.value = !loading.value)
	}

	watch(userData, (newUserData) => {
		if (newUserData) {
			getCollectionData()
		}
	}, { immediate: true })


	return {
		sets: readonly(sets),
		captions: readonly(captions),
		emailSubjectLines: readonly(emailSubjectLines),
		emails: readonly(emails),
		loading: readonly(loading),
		error: readonly(error)
	}
}
